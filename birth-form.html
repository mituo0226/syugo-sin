adjustDayOptions(); validateCurrentInput(); }); console.log('Month event listener added'); } // 日の変更イベント if (daySelect) { daySelect.addEventListener("change", function() { console.log('Day changed to:', this.value); const daySelect = document.getElementById("day"); if (!yearSelect || !monthSelect || !daySelect) { console.log('Elements not found in adjustDayOptions'); return; } const year = parseInt(yearSelect.value); opt.value = d; opt.textContent = d + "日"; daySelect.appendChild(opt); } // 以前選択されていた日が有効な範囲内であれば復元 if (currentDay && parseInt(currentDay) <= daysInMonth) { daySelect.value = currentDay; console.log('Day restored to:', currentDay); } } // 現在の入力を検証する関数 function validateCurrentInput() { const yearSelect = document.getElementById("year"); const monthSelect = document.getElementById("month"); const daySelect = document.getElementById("day"); select.style.boxShadow = "0 0 10px rgba(255, 107, 107, 0.5)"; } }); } // エラーハイライトをクリアする関数 function clearErrorHighlights() { const yearSelect = document.getElementById("year"); const monthSelect = document.getElementById("month"); } }); } // エラー表示をクリアする関数 function clearErrorDisplay() { const errorMessages = document.querySelectorAll(".error-message"); errorMessages.forEach(msg => msg.remove()); clearErrorHighlights(); } // 成功メッセージを表示する関数 function showSuccessMessage() { clearErrorDisplay();
